Cheatsheet

Referência rápida de toda a API em um só lugar. Útil para consultar assinaturas, métodos disponíveis e formatos de retorno sem precisar navegar pela documentação completa.

Funções base #

normalize e validate 
import { normalize, validate } from 'data-handlers'

normalize({ type, value, options })   // string — lança TypeError se inválido
validate({ type, value, options })    // { valid, value, error } — nunca lança

handlers proxy #

handlers — métodos e introspecção 
import { handlers } from 'data-handlers'

handlers.<tipo>.normalize(v, opts)   // string — lança se inválido
handlers.<tipo>.validate(v, opts)    // { valid, value, error } — nunca lança
handlers.<tipo>.parse(v, opts)       // alias de .normalize()
handlers.<tipo>.safe(v, opts)        // alias de .validate()

handlers.has('cpf')                  // boolean — tipo existe no registry?
handlers.types                        // string[] — todos os tipos registrados
handlers.$.has('cpf')               // namespace meta — mesmas funcionalidades
handlers.$.types

Schemas #

schema — definição, métodos e utilitários 
import { schema, SchemaError } from 'data-handlers'

// Definição
const s = schema({
  name:     'name',                                            // forma curta
  doc:      { type: 'cpf', label: 'CPF' },                    // com label
  phone:    { type: 'phone', optional: true },                 // opcional
  country:  { type: 'slug', default: 'brasil' },              // com default
  total:    { type: 'number', options: { style: 'currency', currency: 'BRL' } },
})

// Uso
s.parse(obj)        // objeto normalizado — lança SchemaError com .errors se inválido
s.safeParse(obj)    // { success, data, errors } — nunca lança

// Utilitários (todos retornam novos schemas)
s.partial()              // todos os campos viram opcionais
s.pick('name', 'email') // só esses campos
s.omit('phone')         // sem esse campo
s.extend({ x: 'slug' }) // adiciona campos

Extensão #

register, registerAliases, createPlugin 
import { register, registerAliases, createPlugin } from 'data-handlers'

register('tipo', handler)                  // registra handler customizado
registerAliases('name', 'nome', 'fullName') // múltiplos nomes pro mesmo handler
createPlugin('tipo', handler)               // alias semântico de register()

serve #

data-handlers/serve — tudo disponível 
import {
  serve,           // serve({ port, routes, middleware, auth, rateLimit, websocket, fetch, error })
  systemResponses, // systemResponses[400 | 401 | 403 | 404 | 405 | 429 | 500](req, message?)
  requireAuth,     // Middleware — bloqueia com 401 se req.user é null
  safeCompare,     // safeCompare(a, b) — boolean, tempo constante
} from 'data-handlers/serve'

// Middleware: null → passa, Response → interrompe
// Ordem: [global middleware] → [route middleware] → handler

// Rate limit: sliding window, headers automáticos em todas as respostas
// X-RateLimit-Limit, X-RateLimit-Remaining, X-RateLimit-Reset, Retry-After

// WebSocket: req.upgrade(data) numa rota GET → ws.data em todos os handlers

Handlers do core #

Todos os handlers do core e principais opções 
handlers.name.normalize('joao da silva')              // 'Joao da Silva'
handlers.name.normalize(v, { lowerCaseWords: [] })    // desativa conectivos

handlers.number.normalize(1234.5)                    // '1.234,5'
handlers.number.normalize(v, { style: 'currency', currency: 'BRL' })

handlers.date.normalize(new Date(), { dateStyle: 'long', locale: 'pt-BR' })
handlers.date.normalize('2024-01-15', { dateStyle: 'short' })
handlers.date.normalize(1735689600000)               // timestamp Unix

handlers.password.normalize(v, { minLength: 12, requireNumber: true })

handlers.url.normalize(v)                              // normaliza host para lowercase
handlers.url.normalize(v, { protocols: ['ftp'] })     // habilita ftp

handlers.uuid.normalize(v)                             // lowercase
handlers.uuid.normalize(v, { uppercase: true, version: 4 })

handlers.any.normalize(v)                              // qualquer não-nulo
handlers.any.normalize(v, { demandType: 'number' })   // restringe o primitivo
handlers.any.normalize(v, { transform: (v) => v * 2 }) // callback customizado
